	AuthenticateModal.prototype = {

		init: function () {
			this.listenToCalculator();
			this.listenToCTAs();
			this.listenToHeaderMenuLinks();
			this.fireModalLink.fancybox({
				'showCloseButton': true,
				'hideOnOverlayClick': false,

							"#fancybox-bg-s, #fancybox-bg-sw, #fancybox-bg-w, #fancybox-bg-nw").css("background", "none");
				}
			});
		},

			var self = this;
			//Get calculator data
			self.block.find("input[name=sourceValue]").val(self.sourceAmountInput.val());
			self.block.find("input[name=targetValue]").val(self.targetAmountInput.val());
			self.block.find("input[name=sourceCurrencyId]").val(self.sourceCurrencySelect.val());
			self.block.find("input[name=targetCurrencyId]").val(self.targetCurrencySelect.val());
			self.block.find("input[name=fixType]").val(self.fixTypeInput.val());
		},

		listenToCalculator: function () {
			var self = this;

			this.calculatorForm.submit(function (e) {
				if (typeof exchangeCalculator != "undefined" && exchangeCalculator.options.invalidInput) {
					return false
				}
				e.preventDefault();
				LogUtils.handleTransferInitiation('.event_calc_make_a_transfer', 'click_calculator');

				self.populateFormData();
				self.fireModalLink.click();
			});
		},

		listenToCTAs: function () {
			var self = this;

			this.landingPageCTAs.click(function (e) {
				e.preventDefault();
				self.populateFormData();
				self.fireModalLink.click();
				try{
					LogUtils.trackMixpanelEvent('click header send money cta', {'pageName': 'home page'});
				} catch(er){ console.error('Error sending analytics: ' + er)}

					LogUtils.trackMixpanelEvent('click_header_login', {'pageName': 'home page'});
				} catch(er){ console.error('Error sending analytics: ' + er)}
			});
		}
