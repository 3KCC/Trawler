		var CurrencyWishesHandler = function(block) {
			this.block = block;
			this.curFromSelect = this.block.find('select[name="currencyFrom"]');
			this.curToSelect = this.block.find('select[name="currencyTo"]');
			this.closePopUpLink = this.block.find('a#close-cur-wish-pop');
			this.firePopUpLink = this.block.find('a#fire-cur-wish-pop');

			this.addFriendsForm = this.block.find('#addFriendsForm');
			this.submitStep1 = this.block.find('input.submit-curr-wish');
			this.submitStep2 = this.block.find('input.submit-add-friends');
			this.init();
		};

		CurrencyWishesHandler.prototype = {
			init: function() {
				var self = this;
				this.activeForm = "initial";

				this.closePopUpLink.click(function() {
					$.fancybox.close();
					LogUtils.sendGoogleAnalyticsEvent('CurrencyWishesPopUp', self.activeForm + '-screen-click-close');

					self.resetDialog();
				});

				this.firePopUpLink.fancybox({
					'showCloseButton': false,
					'hideOnOverlayClick' : false,
					'overlayOpacity': 0.5,
					padding: 0,
					onStart: function(){
						$("#fancybox-outer, #fancybox-bg-n, #fancybox-bg-ne, #fancybox-bg-e, #fancybox-bg-se," +
								"#fancybox-bg-s, #fancybox-bg-sw, #fancybox-bg-w, #fancybox-bg-nw").css("background","none");
					}
				});

				this.submitStep1.click(function(e) {
					e.preventDefault();

					var subheadText = self.addFriendsForm.find('.add-friends-subhead').html();

					var currencyFrom = self.curFromSelect.val();
					var currencyTo = self.curToSelect.val();
					var errors = self.curWishListForm.find('.errors');

					$.ajax({
						type: "POST",
						url: "/captureUserData/captureCurrencyWish",
						dataType: "json",
						data: self.curWishListForm.serialize(),
						success: function (result) {
							errors.hide();

							if (result.redirectToPaymentFlow) {
								window.location.href = result.redirectToPaymentFlow;
							}

							if (result.success) {
								self.addFriendsForm.find('.add-friends-subhead').html(Mustache.render(subheadText, {currencyFrom: currencyFrom, currencyTo: currencyTo}));

								self.curWishListForm.hide();
								self.addFriendsForm.show();
								self.activeForm = "final"
							} else {
								if (result.errors) {
									errors.html(self.parseErrorList(result.errors));
									errors.show();
								}
							}
						}
					});

					LogUtils.sendGoogleAnalyticsEvent('CurrencyWishesPopUp', 'initial-screen-all-done-click');
					return false;
				});

				this.submitStep2.click(function(e) {
					e.preventDefault();

					var currencyFrom = self.curFromSelect.val();
					var currencyTo = self.curToSelect.val();
					var emailsInput = self.addFriendsForm.find('#emailsInput');
					var message = self.addFriendsForm.find('.message');
					var errors = self.addFriendsForm.find('.errors');

					$.ajax({
						type: "POST",
						url: "/captureUserData/addFriends",
						dataType: "json",
						data: self.addFriendsForm.serialize() + "&currencyFrom=" + currencyFrom + "&currencyTo=" + currencyTo,
						success: function (result) {
							message.hide();
							errors.hide();
							emailsInput.val('');

								message.show();
								message.text(result.message);
							}

							if (result.errors) {
								errors.html(self.parseErrorList(result.errors));
								errors.show();
							}
						}
					});

					LogUtils.sendGoogleAnalyticsEvent('CurrencyWishesPopUp', 'final-screen-add-friends-to-the-list-click');
					return false;
				});
			},
